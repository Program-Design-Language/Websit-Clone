導関数を求めるVBA関数



Option Explicit

Sub test()
　　Const q = "y = 5x^3 + 1.5x^2.5 + 7x+5"
　　Debug.Print q
　　
　　Dim s As String: s = q
　　If InStr(q, "=") > 0 Then s = Mid(q, InStr(q, "=") + 1)
　　Debug.Print "y' = " & 導関数(s)
End Sub

Function 導関数(q As String) As String
　　Dim s() As String
　　s = Split(空白区切(q), " ")
　　Dim s1
　　Dim i As Long
　　For i = LBound(s) To UBound(s)
　　　　If Not (s(i) = "+" Or s(i) = "-") Then
　　　　　　s(i) = 微分の法則(s(i))
　　　　End If
　　Next
　　導関数 = Trim(Join(s))
　　If Right(導関数, 1) = "+" Or Right(導関数, 1) = "-" Then
　　　　導関数 = Left(導関数, Len(導関数) - 1)
　　End If
　　導関数 = Trim(導関数)
End Function

Function 空白区切(ByVal s As String) As String
　　s = Replace(s, "X", "x")
　　s = Replace(Replace(s, "+", " + "), "-", " - ")
　　s = Replace(s, "^ - ", "^-") '後から追加
　　Do
　　　　s = Replace(s, "　", " ")
　　Loop Until Len(s) = Len(Replace(s, "　", " "))
　　空白区切 = s
End Function

Function 微分の法則(ByVal s As String) As String
　　Dim ary
　　If InStr(s, "x") = 0 Then
　　　　微分の法則 = ""
　　　　Exit Function
　　End If
　　If InStr(s, "^") = 0 Then
　　　　微分の法則 = Replace(s, "x", "")
　　　　Exit Function
　　End If
　　ary = Split(Replace(s, "x", ""), "^")
　　微分の法則 = IIf(ary(0) = "", 1, ary(0)) * ary(1) & _
　　　　　　　　 "x" & _
　　　　　　　　 IIf(ary(1) = 2, "", "^" & ary(1) - 1)
End Function
