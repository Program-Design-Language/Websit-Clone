#Cookie
sub getCookie {
if( $ENV{'HTTP_COOKIE'} ) {
foreach (split(';',$ENV{'HTTP_COOKIE'})){
($name,$value)=split('=');
$value=~ s/%([0-9a-fA-F][0-9a-fA-F])/pack("c",hex($1))/eg;
$pCOOKIE{$name}=$value;
}
}
}
