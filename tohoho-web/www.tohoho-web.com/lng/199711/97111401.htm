@@@@while ( $str =~ m|^([^<]*)(</\w+>)| ) { # ^O
@@@@@@$mstr .= $1; $str = $';

sub find_tag {
@@while ( $str =~ m/^[^<]*<(\w+)[^>]*>/ ) { # O^O
@@@@$mstr .= $&; $str = $';
@@@@($elem = $1) =~ tr/a-z/A-Z/;
@@@@if ( $tags{ $elem } ) { # ^O
