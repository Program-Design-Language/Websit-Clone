if [-f $apache];then

記述した内容が間違っているのでしょうか？

Soralisでしか1.3.6を使用してませんが、
以下のようなかんじでhttpd を直接キックしてます。
参考になればさいわいです。

#!/bin/sh
if  [  -f  /opt/apache/httpd ]; then
/opt/apache/httpd -f /opt/apache/conf/httpd.conf
echo "Starting Apache Web Server version 1.3.6"
fi
