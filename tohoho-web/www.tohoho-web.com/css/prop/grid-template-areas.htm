CSS
.grid-container {
  display: grid;
  width: 30rem;
  height: 15rem;
  grid-template-areas: "header header"
                       "menu main"
                       "footer footer";
  grid-template-rows: 1.5rem auto 1.5rem;
  grid-template-columns: 6rem auto;
}
.box {
  background-color: #fff;
  border: 1px solid #ccc;
  padding: .1rem;
}
.header { grid-area: header; }
.menu { grid-area: menu; }
.main { grid-area: main; }
.footer { grid-area: footer; }
