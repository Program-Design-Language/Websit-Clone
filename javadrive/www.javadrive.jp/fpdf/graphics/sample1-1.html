<?php
require('fpdf/mbfpdf.php');

$pdf=new MBFPDF('P', 'mm', 'A4');
$pdf->AddMBFont(GOTHIC ,'SJIS');
$pdf->AddPage();
$pdf->SetFont(GOTHIC,'',20);

$pdf->Write(10,'直線の描画');
$pdf->Line(10.0, 10.0, 90.0, 10.0);
$pdf->Line(10.0, 10.0, 10.0, 50.0);
$pdf->Line(90.0, 10.0, 10.0, 50.0);

$pdf->Output();
?>
