<?php
require('fpdf/mbfpdf.php');

$pdf=new MBFPDF('P', 'mm', 'A4');
$pdf->AddMBFont(GOTHIC ,'SJIS');
$pdf->AddPage();
$pdf->SetFont(GOTHIC,'',20);

$pdf->SetXY(15.0, 15.0);
$pdf->Write(10,'文字列の出力');

$pdf->SetXY(30.0, 30.0);
$pdf->Write(10,'次の文字列');

$pdf->SetXY(45.0, 45.0);
$pdf->Write(10,'その次の文字列');

$pdf->Output();
?>
