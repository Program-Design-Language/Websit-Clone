    gbc.gridy = 0;

    JButton button2 = new JButton("Yahoo!");

    gbc.gridx = 1;

    gbc.gridheight = 1;
    gbc.weightx = 1.0d;
    gbc.weighty = 1.0d;

    gbc.gridy = 1;
    gbc.weightx = 1.0d;

    p.add(button1);
    p.add(button2);
    p.add(button3);

    getContentPane().add(p, BorderLayout.CENTER);
