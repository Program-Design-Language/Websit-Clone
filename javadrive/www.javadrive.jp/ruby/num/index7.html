т║ЃтЉі

сѓисЃЋсЃѕТ╝ћу«ЌтГљсЂ»сЃЊсЃЃсЃѕТ╝ћу«ЌтГљсЂ«СИђсЂцсЂДТЋ░тђцсЂ«тљёсЃЊсЃЃсЃѕсѓњтидтЈѕсЂ»тЈ│сЂИсѓисЃЋсЃѕсЂЋсЂЏсѓІсЂЪсѓЂсЂ«Т╝ћу«ЌтГљсЂДсЂЎсђѓ

Т╝ћу«ЌтГљУеўУ┐░СЙІТёЈтЉ│

>>a >> bтЈ│сѓисЃЋсЃѕ

сЃЊсЃЃсЃѕТ╝ћу«ЌтГљсЂетљїсЂўсЂЈт»ЙУ▒АсЂ«тђцсѓњ2жђ▓ТЋ░сЂ«сЃЊсЃЃсЃѕсЂесЂЌсЂдУђЃсЂѕсђЂсЃЊсЃЃсЃѕсѓњтидтЈѕсЂ»тЈ│сЂИсѓисЃЋсЃѕсЂЋсЂЏсЂЙсЂЎсђѓсЂфсЂі2жђ▓ТЋ░сЂФсЂісЂёсЂдтидсЂИ1сЂцсЃЊсЃЃсЃѕсѓњсѓисЃЋсЃѕсЂЋсЂЏсѓІсЂЊсЂесЂ»тђцсѓњ2тђЇсЂЎсѓІсЂЊсЂесЂФуЏИтйЊсЂЌсђЂтЈ│сЂИ1сЂцсЃЊсЃЃсЃѕсѓњсѓисЃЋсЃѕсЂЋсЂЏсѓІсЂЊсЂесЂ»тђцсѓњ1/2сЂФсЂЎсѓІсЂЊсЂесЂФуЏИтйЊсЂЌсЂЙсЂЎсђѓ

Рђ╗Т╝ћу«ЌтГљсЂ«тёфтЁѕжаєСйЇсЂФсЂцсЂёсЂдсЂ»сђїТ╝ћу«ЌтГљсЂ«тёфтЁѕжаєСйЇ(жќбС┐ѓТ╝ћу«ЌтГљсђЂУФќуљєТ╝ћу«ЌтГљ)сђЇсѓњтЈѓуЁДсЂЌсЂдСИІсЂЋсЂёсђѓ

тидсѓисЃЋсЃѕ

тидсѓисЃЋсЃѕТ╝ћу«ЌтГљсЂ»тидУЙ║сЂ«ТЋ░тђцсѓњтЈ│УЙ║сЂ«тђцсЂасЂЉтидсЂИсѓисЃЋсЃѕсЂЌсЂЙсЂЎсђѓ 

a << b

СЙІсЂесЂЌсЂд11сѓњ1сЃЊсЃЃсЃѕтидсЂИсѓисЃЋсЃѕсЂЋсЂЏсЂдсЂ┐сЂЙсЂЎсђѓ 

0000000000001011 = 11
-----------------------
0000000000010110 = 22

тљёсЃЊсЃЃсЃѕсѓњТїЄт«џсЂЌсЂЪТЋ░сЂасЂЉтидсЂИсѓисЃЋсЃѕсЂЌсЂЙсЂЎсђѓтидуФ»сЂІсѓЅсЂ»сЂ┐тЄ║сЂЌсЂЪжЃетѕєсЂ»тЅіжЎцсЂЋсѓїсђЂсѓисЃЋсЃѕсЂЌсЂЪсЂЊсЂесЂФсѓѕсЂБсЂдуЕ║сЂёсЂЪтЈ│уФ»сЂ»сђї0сђЇсЂїТа╝у┤ЇсЂЋсѓїсЂЙсЂЎсђѓухљТъюсЂесЂЌсЂд11сѓњ1сЃЊсЃЃсЃѕтидсЂИсѓисЃЋсЃѕсЂЎсѓІсЂе22сЂесЂфсѓісЂЙсЂЎсђѓ 

num1 = 11
num2 = 1
print(num1 << num2)

СИіУеўсЂДсЂ»ухљТъюсЂесЂЌсЂдсђї22сђЇсЂїтЄ║тіЏсЂЋсѓїсЂЙсЂЎсђѓ

тЈ│сѓисЃЋсЃѕ

тЈ│сѓисЃЋсЃѕТ╝ћу«ЌтГљсЂ»тидУЙ║сЂ«ТЋ░тђцсѓњтЈ│УЙ║сЂ«тђцсЂасЂЉтЈ│сЂИсѓисЃЋсЃѕсЂЌсЂЙсЂЎсђѓ 

a >> b

СЙІсЂесЂЌсЂд11сѓњ1сЃЊсЃЃсЃѕтЈ│сЂИсѓисЃЋсЃѕсЂЋсЂЏсЂдсЂ┐сЂЙсЂЎсђѓ 

0000000000001011 = 11
-----------------------
0000000000000101 = 5

тљёсЃЊсЃЃсЃѕсѓњТїЄт«џсЂЌсЂЪТЋ░сЂасЂЉтЈ│сЂИсѓисЃЋсЃѕсЂЌсЂЙсЂЎсђѓтЈ│уФ»сЂІсѓЅсЂ»сЂ┐тЄ║сЂЌсЂЪжЃетѕєсЂ»тЅіжЎцсЂЋсѓїсђЂсѓисЃЋсЃѕсЂЌсЂЪсЂЊсЂесЂФсѓѕсЂБсЂдуЕ║сЂёсЂЪтЈ│уФ»сЂФсЂ»сђЂтЁЃсђЁсЂ«тђцсЂїсђї1сђЇсЂасЂБсЂЪта┤тљѕсЂ»сђї1сђЇсЂїТа╝у┤ЇсЂЋсѓїсђЂсђї0сђЇсЂасЂБсЂЪта┤тљѕсЂ»сђї0сђЇсЂїТа╝у┤ЇсЂЋсѓїсЂЙсЂЎсђѓТЋ░тђцсЂ«угдтЈисЂ»сЂЮсЂ«сЂЙсЂЙсЂесЂёсЂєсЂЊсЂесЂДсЂЎсђѓ 

num1 = 11
num2 = 1
print(num1 >> num2)

СИіУеўсЂДсЂ»ухљТъюсЂесЂЌсЂдсђї5сђЇсЂїтЄ║тіЏсЂЋсѓїсЂЙсЂЎсђѓ

С╗іт║дсЂ»У▓асЂ«сЂДсЂѓсѓІ-10сѓњ1сЃЊсЃЃсЃѕтЈ│сЂИсѓисЃЋсЃѕсЂЋсЂЏсЂдсЂ┐сЂЙсЂЎсђѓ

-----------------------
1111111111111011 = -5

#! ruby -Ku

num2 = 2
print(num1, " << ", num2, " = ", num1 << num2, "¥n")
print(num1, " >> ", num2, " = ", num1 >> num2, "¥n")

num1 = -10
num2 = 1
print(num1, " >> ", num2, " = ", num1 >> num2, "¥n")
